Condicional: if / else

int nota = 8;

if (nota >= 7)
{
    Console.WriteLine("Aprovado!");
}
else if (nota >= 5)
{
    Console.WriteLine("Recuperação");
}
else
{
    Console.WriteLine("Reprovado");
}

// Operador ternário
string resultado = nota >= 7 ? "Aprovado" : "Reprovado";

Condicional: switch

string diaSemana = "segunda";

switch (diaSemana)
{
    case "segunda":
    case "terça":
    case "quarta":
    case "quinta":
    case "sexta":
        Console.WriteLine("Dia útil");
        break;
    case "sábado":
    case "domingo":
        Console.WriteLine("Final de semana!");
        break;
    default:
        Console.WriteLine("Dia inválido");
        break;
}

Repetição: for e while

for

// Contar de 1 a 10
for (int i = 1; i <= 10; i++)
{
    Console.WriteLine(i);
}

// Tabuada do 5
for (int i = 1; i <= 10; i++)
{
    Console.WriteLine(
        $"5 x {i} = {5 * i}");
}

while

// Enquanto condição for verdadeira
int contador = 0;
while (contador < 5)
{
    Console.WriteLine(
        $"Contagem: {contador}");
    contador++;
}

// do-while (executa ao menos 1x)
do
{
    Console.Write("Opção: ");
} while (opcao != 0);

Arrays e Listas

Array (tamanho fixo)

// Declaração
string[] frutas = { "Maçã", 
    "Banana", "Laranja" };

// Acesso por índice
Console.WriteLine(frutas[0]);
// "Maçã"

// Percorrer
foreach (var f in frutas)
{
    Console.WriteLine(f);
}

List<T> (tamanho dinâmico)

// Declaração
List<string> alunos = new()
{
    "Maria", "João", "Ana"
};

// Adicionar
alunos.Add("Pedro");

// Remover
alunos.Remove("João");

// Quantidade
alunos.Count; // 3

Usando Listas nas Views Razor

Controller

public IActionResult Alunos()
{
    var lista = new List<string>
    {
        "Maria", "João",
        "Ana", "Pedro"
    };
    ViewBag.Alunos = lista;
    return View();
}

View (Alunos.cshtml)

<h2>Lista de Alunos</h2>
<ul>
@foreach (var aluno 
    in ViewBag.Alunos)
{
    <li>@aluno</li>
}
</ul>
<p>Total: 
  @ViewBag.Alunos.Count
</p>

Exercício Prático 1

Tabela de Multiplicação no MVC

  1. Crie um Controller chamado TabelaController
  2. A Action Index(int numero) recebe um número como parâmetro
  3. Passe o número via ViewBag para a View
  4. Na View, use @for para gerar a tabuada de 1 a 10
  5. Exiba em uma tabela HTML estilizada
  6. Teste acessando: /Tabela/Index/7

Exercício Prático 2

Lista de Tarefas (To-Do List)

  1. Crie um Controller TarefaController
  2. Na Action Index, crie uma List<string> com 5 tarefas
  3. Passe a lista via ViewBag para a View
  4. Na View, exiba as tarefas em uma lista HTML
  5. Use @if para verificar se a lista está vazia
  6. Numere as tarefas usando o índice do loop
  7. Estilize com CSS (cores alternadas nas linhas)
